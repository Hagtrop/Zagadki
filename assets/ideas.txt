[3]
Реализовать таймер. См. http://stackoverflow.com/questions/4597690/android-timer-how

Handler handler = new Handler();
MyTimer timer = new MyTimer(handler, tv1);
handler.postDelayed(timer, 0);

class MyTimer implements Runnable{
	Handler h;
	TextView tv;
	
	MyTimer(Handler h, TextView tv){
		this.tv = tv;
		this.h = h;
	}

	@Override
	public void run() {
		tv.setText(String.valueOf(System.currentTimeMillis()));
		h.postDelayed(this, 500);
	}
}
